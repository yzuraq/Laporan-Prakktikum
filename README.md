# Laporan-Prakktikum
Modul 1 dan 2

        // Loop for
        for (i = 1; i <= 10; i++) {
            System.out.println(Integer.toString(i));
        }

        // Loop while
        i = 1;
        while (i <= 10) {
            System.out.println(Integer.toString(i));
            i++;
        }

        // Loop do-while
        i = 1;
        do {
            System.out.println(Integer.toString(i));
            i++;
        } while (i <= 10);
